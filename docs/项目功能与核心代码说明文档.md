## 一、航班信息展示与搜索功能

```java
@Service
public class FlightService {

    private final FlightRepository flightRepository;

    public FlightService(FlightRepository flightRepository) {
        this.flightRepository = flightRepository;
    }

    public List<Flight> getAllFlights(boolean includePast) {
        List<Flight> flights;
        if (includePast) {
            flights = flightRepository.findAllByOrderByDepartureTimeAsc();
        } else {
            flights = flightRepository.findAvailableFlights();
        }
        return prepareFlightList(flights, includePast);
    }

    public List<Flight> searchFlights(FlightSearchRequest request, boolean includePast) {
        String originKeyword = normalizeKeyword(request.getOrigin());
        String destinationKeyword = normalizeKeyword(request.getDestination());

        List<Flight> flights;
        if (originKeyword != null && destinationKeyword != null) {
            flights = flightRepository
                    .findByOriginContainingIgnoreCaseAndDestinationContainingIgnoreCase(
                            originKeyword,
                            destinationKeyword
                    );
        } else if (originKeyword != null) {
            flights = flightRepository.findByOriginContainingIgnoreCase(originKeyword);
        } else if (destinationKeyword != null) {
            flights = flightRepository.findByDestinationContainingIgnoreCase(destinationKeyword);
        } else {
            flights = flightRepository.findAvailableFlights();
        }
        return prepareFlightList(flights, includePast);
    }

    public List<String> searchCities(String keyword, int limit) {

        }

        if (trimmed.isEmpty()) {

    }
```

```ts
import apiClient from "./api";

export interface FlightQuery {
  date?: string;
  from?: string;
  to?: string;
}

export interface FlightSeat {
  id?: number;
  cabinClass: string;
  price: number;
  remainingSeats: number;
  totalSeats: number;
}

export interface Flight {
  id: string;
  flightId: number;
  flightNumber: string;
  airline: string;
  aircraftType?: string;
  from: string;
  to: string;
  departTime: string;
  arriveTime: string;
  duration: string;
  price: number;
  seatsLeft: number;
  seats?: FlightSeat[];
  planeType?: string;
  status?: "ON_TIME" | "DELAYED" | "CANCELLED" | string;
  remarks?: string;
}

interface FlightApiEntity {
  id?: number | string;
  flightNumber?: string;
  airline?: string;
  aircraftType?: string;
  origin?: string;
  destination?: string;
  departureTime?: string;
  arrivalTime?: string;
  duration?: string;
  price?: number;
  seatsAvailable?: number;
  remainingSeats?: number;
  seats?: FlightSeatEntity[];
  planeType?: string;
  status?: string;
  remarks?: string;
}

const mapFlight = (data: FlightApiEntity): Flight => {
  const departTime = data.departureTime || new Date().toISOString();
  const arriveTime = data.arrivalTime || departTime;

  let totalRemainingSeats = 0;
  let minPrice = 0;

  if (data.seats && Array.isArray(data.seats) && data.seats.length > 0) {
    totalRemainingSeats = data.seats.reduce((sum, seat) => {
      return sum + (Number(seat.remainingSeats) || 0);
    }, 0);

    const prices = data.seats
      .map((seat) => Number(seat.price || 0))
      .filter((price) => price > 0);
    minPrice = prices.length > 0 ? Math.min(...prices) : 0;
  } else {
    totalRemainingSeats = Number(data.seatsAvailable ?? data.remainingSeats ?? 0);
    minPrice = Number(data.price ?? 0);
  }

  return {
    id: data.flightNumber || String(data.id ?? Date.now()),
    flightId: Number(data.id ?? 0),
    flightNumber: data.flightNumber || "未知航班",
    airline: data.airline || "",
    aircraftType: data.aircraftType,
    from: data.origin || "未知出发地",
    to: data.destination || "未知目的地",
    departTime,
    arriveTime,
    duration: data.duration || "",
    price: minPrice,
    seatsLeft: totalRemainingSeats,
    seats: data.seats?.map((seat) => ({
      id: seat.id,
      cabinClass: seat.cabinClass || "",
      price: Number(seat.price || 0),
      remainingSeats: Number(seat.remainingSeats || 0),
      totalSeats: Number(seat.totalSeats || 0),
    })),
    planeType: data.planeType,
    status: (data.status as Flight["status"]) || "ON_TIME",
    remarks: data.remarks,
  };
};

export const flightApi = {
  async listAll(options: { includePast?: boolean } = {}): Promise<Flight[]> {
    const params = options.includePast ? { includePast: options.includePast } : undefined;
    const resp = await apiClient.get("/api/flights", { params });
    return normalizeResponse(resp.data).map(mapFlight);
  },

  async search(query: FlightQuery = {}): Promise<Flight[]> {
    const payload: Record<string, string> = {};
    if (query.from?.trim()) {
      payload.origin = query.from.trim();
    }
    if (query.to?.trim()) {
      payload.destination = query.to.trim();
    }
    const formattedDate = formatDateTime(query.date);
    if (formattedDate) {
      payload.departureDate = formattedDate;
    }
    const response = await apiClient.post("/api/flights/search", payload);
    return normalizeResponse(response.data).map(mapFlight);
  },

  async searchCities(keyword = "", limit = 20): Promise<string[]> {
    const params: Record<string, string | number> = { limit };
    const trimmed = keyword.trim();
    if (trimmed.length > 0) {
      params.keyword = trimmed;
    }
    const resp = await apiClient.get("/api/flights/cities", { params });
    return normalizeResponse(resp.data);
  },
};
```

```java
@Service
public class UserService {

    private final UserRepository userRepository;
    private final PasswordEncoder passwordEncoder;

    public UserService(UserRepository userRepository, PasswordEncoder passwordEncoder) {
        this.userRepository = userRepository;
        this.passwordEncoder = passwordEncoder;
    }

    public User register(RegisterRequest request) {
        if (userRepository.findByUsername(request.getUsername()).isPresent()) {
            throw new RuntimeException("用户名已存在");
        }
        if (userRepository.findByIdNumber(request.getIdNumber()).isPresent()) {
            throw new RuntimeException("身份证号已被使用");
        }
        if (userRepository.findByPhone(request.getPhone()).isPresent()) {
            throw new RuntimeException("手机号已被使用");
        }

        User user = new User();
        user.setUsername(request.getUsername());
        user.setPassword(passwordEncoder.encode(request.getPassword()));
        user.setName(request.getName());
        user.setIdNumber(request.getIdNumber());
        user.setPhone(request.getPhone());
        user.setAddress(request.getAddress());
        user.setRole("USER");
        user.setStatus(1);
        user.setWalletBalance(BigDecimal.valueOf(200));
        return userRepository.save(user);
    }

    public User login(LoginRequest request) {
        User user = userRepository.findByUsername(request.getUsername())
                .orElseThrow(() -> new RuntimeException("用户名或密码错误"));

        if (!passwordEncoder.matches(request.getPassword(), user.getPassword())) {
            throw new RuntimeException("用户名或密码错误");
        }

        if (user.getWalletBalance() == null) {
            user.setWalletBalance(BigDecimal.valueOf(200));
            userRepository.save(user);
        }

        return user;
    }
}
```

```ts
export const saveAuth = (userInfo: UserInfo) => {
  if (typeof window !== "undefined") {
    localStorage.setItem(AUTH_STORAGE_KEY, JSON.stringify(userInfo));
  }
};

export const getAuth = (): UserInfo | null => {
  if (typeof window === "undefined") {
    return null;
  }
  try {
    const authData = localStorage.getItem(AUTH_STORAGE_KEY);
    return authData ? JSON.parse(authData) : null;
  } catch (error) {
    console.error("获取认证信息失败:", error);
    return null;
  }
};

export const clearAuth = () => {
  if (typeof window !== "undefined") {
    localStorage.removeItem(AUTH_STORAGE_KEY);
  }
};

export const isAuthenticated = (): boolean => {
  const auth = getAuth();
  return auth !== null && !!auth.token;
};

export const getToken = (): string | null => {
  const auth = getAuth();
  return auth?.token || null;
};
```

### 3.3 后端核心代码：BookingService#createBooking（节选）

```java
@Service
@Transactional
public class BookingService {

    private final BookingRepository bookingRepository;
    private final UserRepository userRepository;
    private final FlightRepository flightRepository;
    private final SeatService seatService;

    public BookingService(BookingRepository bookingRepository,
                          UserRepository userRepository,
                          FlightRepository flightRepository,
                          SeatService seatService) {
        this.bookingRepository = bookingRepository;
        this.userRepository = userRepository;
        this.flightRepository = flightRepository;
        this.seatService = seatService;
    }

    public Booking createBooking(BookingRequest request) {
        User user = userRepository.findById(request.getUserId())
                .orElseThrow(() -> new RuntimeException("用户不存在"));
        Flight flight = flightRepository.findById(request.getFlightId())
                .orElseThrow(() -> new RuntimeException("航班不存在"));

        validatePassengerIdentities(request);

        int remainingSeats = calculateRemainingSeatsFromSeatTable(
                request.getFlightId(), request.getCabinClass());
        if (remainingSeats < request.getTicketCount()) {
            throw new RuntimeException("该舱位剩余座位不足");
        }

        BigDecimal cabinPrice = findCabinPrice(flight, request.getCabinClass());
        BigDecimal baseAmount = cabinPrice.multiply(
                BigDecimal.valueOf(request.getTicketCount()));

        boolean urgentDeparture = isWithinUrgentWindow(flight.getDepartureTime());
        BigDecimal surchargeRate = urgentDeparture
                ? BookingRuleConstants.URGENT_SURCHARGE_RATE
                : BigDecimal.ZERO;

        BigDecimal totalAmount = baseAmount
                .add(baseAmount.multiply(surchargeRate))
                .setScale(2, RoundingMode.HALF_UP);

        if (!Boolean.TRUE.equals(request.getPayLater())) {
            BigDecimal walletBalance = user.getWalletBalance();
            if (walletBalance.compareTo(totalAmount) < 0) {
                throw new RuntimeException("钱包余额不足，请先充值或减少购票数量");
            }
        }

        Booking booking = new Booking();
        booking.setFlight(flight);
        booking.setUser(user);
        booking.setCabinClass(request.getCabinClass());
        booking.setTicketCount(request.getTicketCount());
        booking.setTotalAmount(totalAmount);
        booking.setUrgentSurchargeRate(surchargeRate);

        if (Boolean.TRUE.equals(request.getPayLater())) {
            booking.setStatus("CREATED");
            booking.setPaymentDueAt(LocalDateTime.now().plusMinutes(15));
        } else {
            booking.setStatus("PAID");
        }

        Booking saved = bookingRepository.save(booking);

        SeatSelectionRequest seatRequest = new SeatSelectionRequest();
        seatRequest.setFlightId(request.getFlightId());
        seatRequest.setCabinClass(request.getCabinClass());
        seatRequest.setTicketCount(request.getTicketCount());
        seatRequest.setSelectedColumns(request.getSelectedSeatColumns());

        SeatSelectionResponse seatResponse = seatService.selectSeats(seatRequest);
        saveBookingSeats(saved, seatResponse, request);

        if (!Boolean.TRUE.equals(request.getPayLater())) {
            user.setWalletBalance(user.getWalletBalance().subtract(totalAmount));
            userRepository.save(user);
        }

        return saved;
    }

    private void validatePassengerIdentities(BookingRequest request) {
        Set<String> idNumbers = new HashSet<>();
        Set<String> phones = new HashSet<>();
        for (BookingRequest.PassengerInfo passenger : request.getPassengers()) {
            String idNumber = passenger.getIdNumber();
            String phone = passenger.getPhone();
            if (!idNumbers.add(idNumber)) {
                throw new RuntimeException("身份证号 " + idNumber + " 已被重复使用");
            }
            if (!phones.add(phone)) {
                throw new RuntimeException("联系电话 " + phone + " 已被重复使用");
            }
        }
    }
}
```

## 四、座位选择与分配功能

### 4.1 功能说明

- 用于在新座位系统下按“列”（A/B/C/D 等）为乘客分配具体座位号；
- 在数据库层面通过 `seats` + `booking_seats` 表记录座位占用；
- 通过事务 + 实时查询避免同一座位被重复分配。

### 4.2 后端核心代码：SeatService（节选）

```java
@Service
public class SeatService {

    private final SeatRepository seatRepository;
    private final BookingSeatRepository bookingSeatRepository;

    public SeatService(SeatRepository seatRepository,
                       BookingSeatRepository bookingSeatRepository) {
        this.seatRepository = seatRepository;
        this.bookingSeatRepository = bookingSeatRepository;
    }

    public SeatSelectionResponse getAvailableSeatColumns(Long flightId, String cabinClass) {
        List<Seat> allSeats = seatRepository.findByFlightIdAndCabinClass(flightId, cabinClass);
        Set<Long> occupiedSeatIds = loadOccupiedSeatIds(flightId, cabinClass);

        Map<String, Long> availableCountMap = new HashMap<>();
        for (Seat seat : allSeats) {
            String column = seat.getSeatColumn();
            boolean seatAvailable = Boolean.TRUE.equals(seat.getIsAvailable())
                    && !occupiedSeatIds.contains(seat.getId());
            if (seatAvailable) {
                availableCountMap.put(column,
                        availableCountMap.getOrDefault(column, 0L) + 1);
            }
        }

        SeatSelectionResponse response = new SeatSelectionResponse();
        response.setAvailableColumns(new ArrayList<>(availableCountMap.keySet()));
        // ... 组装 columnAvailability 略
        return response;
    }

    @Transactional
    public SeatSelectionResponse selectSeats(SeatSelectionRequest request) {
        Long flightId = request.getFlightId();
        String cabinClass = request.getCabinClass();
        int ticketCount = request.getTicketCount();
        List<String> selectedColumns = request.getSelectedColumns();

        Set<Long> occupiedSeatIds = loadOccupiedSeatIds(flightId, cabinClass);
        List<String> assignedSeats = new ArrayList<>();
        List<String> unavailableColumns = new ArrayList<>();

        for (String column : selectedColumns) {
            List<Seat> columnSeats = seatRepository
                    .findAvailableSeatsByColumn(flightId, cabinClass, column);
            long availableCount = columnSeats.stream()
                    .filter(seat -> !occupiedSeatIds.contains(seat.getId()))
                    .count();
            if (availableCount == 0) {
                unavailableColumns.add(column);
            }
        }

        List<String> columnsToAssign = buildColumnsToAssign(selectedColumns, ticketCount);
        for (String column : columnsToAssign) {
            List<Seat> columnSeats = seatRepository
                    .findAvailableSeatsByColumn(flightId, cabinClass, column);
            Optional<Seat> availableSeat = columnSeats.stream()
                    .filter(seat -> !occupiedSeatIds.contains(seat.getId()))
                    .findFirst();
            if (availableSeat.isPresent()) {
                Seat seat = availableSeat.get();
                assignedSeats.add(seat.getSeatNumber());
                occupiedSeatIds.add(seat.getId());
            }
        }

        SeatSelectionResponse response = new SeatSelectionResponse();
        response.setAssignedSeats(assignedSeats);
        response.setUnavailableColumns(unavailableColumns);
        return response;
    }
}
```

```java
public Booking payBooking(Long bookingId, Long userId) {
    Booking booking = bookingRepository.findById(bookingId)
            .orElseThrow(() -> new RuntimeException("订单不存在"));
    if (!Objects.equals(booking.getUser().getId(), userId)) {
        throw new RuntimeException("无权支付该订单");

        throw new RuntimeException("钱包余额不足");
    }

    user.setWalletBalance(walletBalance.subtract(booking.getTotalAmount()));
    booking.setStatus("PAID");
    booking.setPaymentDueAt(null);

    userRepository.save(user);
    return bookingRepository.save(booking);
}
```

```java
public Booking refundBooking(Long bookingId, String reason) {
    Booking booking = bookingRepository.findById(bookingId)
        .orElseThrow(() -> new RuntimeException("订单不存在"));

    if (!"PAID".equals(booking.getStatus())) {
        throw new RuntimeException("只有已支付的订单可以退票");
    }

    boolean needsReview = shouldRequireRefundReview(booking);

    if (needsReview) {
        booking.setStatus("REFUND_REVIEW");
        booking.setRefundReason(reason);
    } else {
        processRefund(booking, reason);
    }

    return bookingRepository.save(booking);
}

private void processRefund(Booking booking, String reason) {
    BigDecimal refundAmount = calculateRefundAmount(booking);
    User user = booking.getUser();
    user.setWalletBalance(user.getWalletBalance().add(refundAmount));
    userRepository.save(user);

    releaseSeats(booking);

    booking.setStatus("CANCELED");
    booking.setRefundReason(reason);
}

private BigDecimal calculateRefundAmount(Booking booking) {
    BigDecimal totalAmount = booking.getTotalAmount();
    LocalDateTime now = LocalDateTime.now();
    LocalDateTime departureTime = booking.getFlight().getDepartureTime();

    if (now.isAfter(departureTime.minusHours(12))) {
        BigDecimal refundFeeRate = BookingRuleConstants.URGENT_REFUND_FEE_RATE;
        BigDecimal refundFee = totalAmount.multiply(refundFeeRate);
        booking.setRefundFeeRate(refundFeeRate);
        booking.setRefundFeeAmount(refundFee);
        return totalAmount.subtract(refundFee);
    } else {
        booking.setRefundFeeRate(BigDecimal.ZERO);
        booking.setRefundFeeAmount(BigDecimal.ZERO);
        return totalAmount;
    }
}
```

```java
public Booking changeBooking(Long bookingId, ChangeBookingRequest request) {
    Booking originalBooking = bookingRepository.findById(bookingId)
        .orElseThrow(() -> new RuntimeException("订单不存在"));

    if (!"PAID".equals(originalBooking.getStatus())) {
        throw new RuntimeException("只有已支付的订单可以改签");
    }

    LocalDateTime now = LocalDateTime.now();
    if (now.isAfter(originalBooking.getFlight().getDepartureTime().minusHours(2))) {
        throw new RuntimeException("改签必须在航班起飞前2小时进行");
    }

    Flight newFlight = flightRepository.findById(request.getNewFlightId())
        .orElseThrow(() -> new RuntimeException("新航班不存在"));

    if (Objects.equals(newFlight.getId(), originalBooking.getFlight().getId())
        && request.getNewCabinClass().equalsIgnoreCase(originalBooking.getCabinClass())) {
        throw new RuntimeException("不能改签到同一航班的相同舱位");
    }

    ChangePriceResponse priceResponse = calculateChangePrice(bookingId, request);

    Booking newBooking = new Booking();
    // ... 设置新订单字段（用户、航班、舱位、金额等）
    newBooking.setOriginalBookingId(originalBooking.getId());
    newBooking.setChangeFeeAmount(priceResponse.getChangeFee());

    originalBooking.setStatus("CANCELED");

    return bookingRepository.save(newBooking);
}
```
